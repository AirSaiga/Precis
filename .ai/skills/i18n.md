## Vue 组件中使用

```vue
<template>
  <div class="toolbar">
    <button>{{ t('canvas.toolbar.addNode') }}</button>
    <button>{{ t('canvas.toolbar.validate') }}</button>
    
    <!-- 带插值的翻译 -->
    <span>
      {{ t('validation.result.errors', { count: errorCount }) }}
    </span>
    
    <!-- 复数处理 -->
    <span>
      {{ t('validation.result.itemCount', errorCount, { count: errorCount }) }}
    </span>
  </div>
</template>

<script setup lang="ts">
import { useI18n } from 'vue-i18n'

const { t, locale } = useI18n()

// 动态切换语言
function switchLanguage(lang: string) {
  locale.value = lang
  localStorage.setItem('precis-language', lang)
}
</script>
```

## 组合式函数中的国际化

```typescript
// composables/useValidationMessages.ts
import { useI18n } from 'vue-i18n'
import type { ValidationResult } from '@/types/validation'

export function useValidationMessages() {
  const { t } = useI18n()
  
  function formatResult(result: ValidationResult): string {
    if (result.success) {
      return t('validation.result.passed')
    }
    
    const errorCount = result.errors?.length ?? 0
    const warningCount = result.warnings?.length ?? 0
    
    if (errorCount > 0) {
      return t('validation.result.errors', { count: errorCount })
    }
    
    return t('validation.result.warnings', { count: warningCount })
  }
  
  function getConstraintTypeName(type: string): string {
    return t(`constraint.types.${type}`, type)
  }
  
  return { formatResult, getConstraintTypeName }
}
```

## 新增文案流程

1. **先在代码中使用翻译键**（即使中文环境还没加）
2. **同步更新所有语言文件**
3. **避免硬编码字符串**

```typescript
// ❌ 错误：硬编码
const label = '用户邮箱'

// ✅ 正确：使用翻译键
const label = t('constraint.fields.email')
```

## 后端错误消息国际化

后端返回错误码，前端根据错误码显示对应的翻译：

```python
# backend/app/api/middleware/exception_handler.py
from fastapi import Request
from fastapi.responses import JSONResponse

ERROR_MESSAGES = {
    "zh-CN": {
        "PROJECT_NOT_FOUND": "项目不存在",
        "VALIDATION_ERROR": "请求参数校验失败",
        "AI_PROVIDER_ERROR": "AI 服务调用失败"
    },
    "en-US": {
        "PROJECT_NOT_FOUND": "Project not found",
        "VALIDATION_ERROR": "Request validation failed",
        "AI_PROVIDER_ERROR": "AI service call failed"
    }
}

async def internationalized_exception_handler(request: Request, exc: APIError):
    """国际化错误处理"""
    # 从请求头获取语言偏好
    lang = request.headers.get('Accept-Language', 'zh-CN').split(',')[0]
    
    # 获取本地化消息
    message = ERROR_MESSAGES.get(lang, ERROR_MESSAGES['zh-CN']).get(
        exc.error_code, 
        exc.detail
    )
    
    return JSONResponse(
        status_code=exc.status_code,
        content={
            "success": False,
            "error": {
                "code": exc.error_code,
                "message": message,
                "details": exc.extra_data
            }
        }
    )
```

前端根据错误码自动翻译：

```typescript
// api/errorHandler.ts
import { useI18n } from 'vue-i18n'

export function handleApiError(error: ApiError) {
  const { t } = useI18n()
  
  // 优先使用前端本地化的错误码翻译
  const message = t(`errors.${error.code}`, error.message)
  
  showToast({
    type: 'error',
    message
  })
}
```

## 日期与数字格式化

```typescript
import { useI18n } from 'vue-i18n'

const { d, n } = useI18n()

// 日期
const formattedDate = d(new Date(), 'short')  // 2024/01/15
const formattedDateTime = d(new Date(), 'long')  // 2024年1月15日 14:30

// 数字
const formattedNumber = n(1234567.89)  // 1,234,567.89 (en) / 1 234 567,89 (zh)
const formattedPercent = n(0.8567, { style: 'percent' })  // 85.67%
```

## 语言切换持久化

```typescript
// i18n/index.ts
import { createI18n } from 'vue-i18n'
import zhCN from './locales/zh-CN'
import enUS from './locales/en-US'

const savedLocale = localStorage.getItem('precis-language') || 'zh-CN'

export const i18n = createI18n({
  legacy: false,
  locale: savedLocale,
  fallbackLocale: 'zh-CN',
  messages: {
    'zh-CN': zhCN,
    'en-US': enUS
  }
})
```

## 维护检查清单

- [ ] 新增 UI 文案时，同步更新 `zh-CN.ts` 和 `en-US.ts`
- [ ] 翻译键采用分层命名（`domain.subdomain.key`）
- [ ] 不在模板中硬编码中文或英文
- [ ] 日期/数字使用 `d()` / `n()` 格式化
- [ ] 后端错误返回错误码，前端负责本地化显示
- [ ] 定期清理不再使用的翻译键
